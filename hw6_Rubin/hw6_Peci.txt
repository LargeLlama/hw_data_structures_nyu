fix(u)

		fix(u.right)
	if (u.left != null && u.left < u)
		fix(u.left)

		fix(u.parent)

		fix(u.parent)

check(u)
	Node turtle = head;
	Node rabbit = head;
	while(turtle != null && rabbit != null && rabbit.next != null)
		turtle = turtle.next;
		rabbit = rabbit.next.next;
		if(turtle == rabbit)
			return false
	return true
